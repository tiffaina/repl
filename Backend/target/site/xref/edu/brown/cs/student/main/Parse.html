12  /**
13   * Parse class parses a reader object and saves the object as a list of rows, with the row type
14   * depending on the CreatorFromRow instance that is used.
15   *
16   * @param <T> A generic type parameter, so that the dataRows object can be generic, according to a
17   *     strategy pattern.
18   */
19  public class Parse<T> {
20    private List<T> dataRows;
21    static final Pattern regexSplitCSVRow =
22        Pattern.compile(",(?=([^\\\"]*\\\"[^\\\"]*\\\")*(?![^\\\"]*\\\"))");
23  
24    /**
25     * Constructor to parse a reader object.
26     *
27     * @param reader the object from which to read in data
28     * @param rowType the type of object each row should be converted into
29     * @param hasHeader a boolean of whether the object has a header
30     */
31    public Parse(Reader reader, CreatorFromRow<T> rowType, boolean hasHeader) {
32      this.dataRows = new ArrayList<>();
33      this.parse(reader, rowType, hasHeader);
34    }
35  
36    /**
37     * Parse method reads in a reader object, converting each row to a list of strings, and then
38     * converting each row to whatever the create method returns. The objects returned are added to
39     * dataRows which represents the parsed reader object.
40     */
41    public void parse(Reader reader, CreatorFromRow<T> creatorFromRow, boolean hasHeader) {
42      try {
43        // read the csv
44        BufferedReader bReader = new BufferedReader(reader);
45        /* Go through each line in the reader object, process it, & add it to dataRows. */
46        String line;
47        int counter = 0;
48        while ((line = bReader.readLine()) != null) {
49          // breaks row into an array of strings
50          List<String> rowData = Arrays.asList(regexSplitCSVRow.split(line));
51          // if there is a header, save it to dataRows without converting it
52          // through the create method
53          if (hasHeader) {
54            if (counter == 0) {
55              T header = (T) rowData;
56              this.dataRows.add(header);
57              counter++;
58              continue;
59            }
60          }
61  
62          try {
63            // process the rest of the rows using create() & add them to dataRows
64            T processedRow = creatorFromRow.create(rowData);
65            this.dataRows.add(processedRow);
66            counter++;
67          } catch (CreatorFromRow.FactoryFailureException e) {
68            System.err.println(
69                "Factory Failure Exception occurred. Row cannot be "
70                    + "added to the parsed object"
71                    + rowData);
72            System.exit(1);
73          }
74        }
75      } catch (IOException e) {
76        System.err.println("An IO Exception occurred");
77        System.exit(1);
78      }
79    }
80  
81    /**
82     * getDataRows() is a simple getter for dataRows.
83     *
84     * @return dataRows, the object that represents the parsed reader object
85     */
86    public List<T> getDataRows() {
87      return this.dataRows;
88    }
89  }
