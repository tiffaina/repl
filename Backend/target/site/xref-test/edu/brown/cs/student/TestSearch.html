86      List<List<String>> expectedResults = new ArrayList<>(List.of(innerList));
87      Assert.assertTrue(searchResults1.equals(expectedResults));
88  
89      // shows that there are no matches when the wrong col is searched w/ colName
90      // implementation

102   /**
103    * This tests that out of bounds errors are called as expected when various indexes are passed
104    * into search
105    */
106   @Test
107   public void testOutOfBoundsErrors() {
108     // case 1: colIdx > num columns
109     boolean idxTooLarge = false;
110     try {
111       Search searcher = new Search();
112       searcher.search("data/stars/ten-star.csv", "Sol", 20);
113     } catch (IndexOutOfBoundsException e) {
114       idxTooLarge = true;
115     }
116     Assert.assertTrue(idxTooLarge);
117 
118     // case 2: colIdx < 0
119     boolean idxBelowZero = false;
120     try {
121       Search searcher = new Search();
122       searcher.search("data/stars/ten-star.csv", "Sol", -5);
123     } catch (IndexOutOfBoundsException e) {
124       idxBelowZero = true;
125     }
126     Assert.assertTrue(idxBelowZero);
127 
128     // case 3: correct input for colIdx
129     boolean idxInRange = true;
130     try {
131       Search searcher = new Search();
132       searcher.search("data/stars/ten-star.csv", "Sol", 3);
133     } catch (IndexOutOfBoundsException e) {
134       idxInRange = false;
135     }
136     Assert.assertTrue(idxInRange);
137   }
