3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * IntegerList class implements the CreatorFromRow interface and transforms rows into lists of
8    * integers.
9    */
10  public class IntegerList implements CreatorFromRow<List<Integer>> {
11  
12    /**
13     * Converts all the objects in a row to integers.
14     *
15     * @param row the row to convert to a list of integers
16     * @return the row represented as a list of integers
17     */
18    @Override
19    public List<Integer> create(List<String> row) {
20      System.out.println("running create method in IntegerList");
21      List<Integer> convertedRow = new ArrayList<>();
22      for (String val : row) {
23        try {
24          Integer convertedVal = Integer.valueOf(val);
25          convertedRow.add(convertedVal);
26        } catch (ClassCastException e) {
27          System.err.println(
28              val
29                  + " cannot be converted to an integer. Please "
30                  + "make sure that your CSV includes purely numeric data outside"
31                  + "of the header when trying to parse it into a list of integers.");
32          System.exit(1);
33        } catch (NumberFormatException e) {
34          System.err.println(
35              val
36                  + " cannot be converted to an integer. Please "
37                  + "make sure that your CSV contains integers without decimal "
38                  + "values. ");
39          System.exit(1);
40        }
41      }
42      return convertedRow;
43    }
44  }
