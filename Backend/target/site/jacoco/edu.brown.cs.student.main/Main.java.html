package edu.brown.cs.student.main;

/** The Main class of our project. This is where execution begins. */
public final class Main {
  /**
   * The initial method called when execution begins.
   *
   * @param args An array of command line arguments
   */
  public static void main(String[] args) {
    new Main(args).run();
  }

  private Main(String[] args) {}

  private void run() {
    Search searcher = new Search();
    // example w/out column identifier
    searcher.search("data/census/dol_ri_earnings_disparity.csv", "Black", true);

    // example with colName passed in
    searcher.search("data/census/dol_ri_earnings_disparity.csv", "Black", "Data Type");

    // example with colIdx specified
    searcher.search("data/census/dol_ri_earnings_disparity.csv", "Black", 1);
  }
}
